#  Inicializar tabla de eliminadas 
eliminadas_inf <- data.frame(
  Variable = character(),
  VIF = character(),
  stringsAsFactors = FALSE
)

#  Eliminar filas con NA (recomendado para cor/vif/lm) 
datos_numericos_limpio_3 <- datos_numericos_limpio_3[complete.cases(datos_numericos_limpio_3), ]

#  Eliminar columnas con varianza casi cero 
nzv <- nearZeroVar(datos_numericos_limpio_3)
if (length(nzv) > 0) {
  eliminadas_inf <- bind_rows(
    eliminadas_inf,
    data.frame(Variable = names(datos_numericos_limpio_3)[nzv], VIF = NA)
  )
  datos_numericos_limpio_3 <- datos_numericos_limpio_3[, -nzv, drop = FALSE]
}

#  Eliminar columnas duplicadas por nombre 
duplicadas <- duplicated(names(datos_numericos_limpio_3))
if (any(duplicadas)) {
  eliminadas_inf <- bind_rows(
    eliminadas_inf,
    data.frame(Variable = names(datos_numericos_limpio_3)[duplicadas], VIF = NA)
  )
  datos_numericos_limpio_3 <- datos_numericos_limpio_3[, !duplicadas, drop = FALSE]
}

#  Seleccionar predictores excluyendo Lim_inf 
stopifnot("Lim_inf" %in% names(datos_numericos_limpio_3))

predictores_inf <- datos_numericos_limpio_3[, !(names(datos_numericos_limpio_3) %in% "Lim_inf"), drop = FALSE]

# Asegurar solo numéricos (crítico para cor y vifstep) 
predictores_inf <- predictores_inf[, sapply(predictores_inf, is.numeric), drop = FALSE]

#  Eliminar variables altamente correlacionadas (usa un cutoff útil) 
cor_matrix <- cor(predictores_inf, use = "pairwise.complete.obs")
high_corr <- findCorrelation(cor_matrix, cutoff = 0.95)  # cambia a 0.90 si tienes muchos índices similares

if (length(high_corr) > 0) {
  eliminadas_inf <- bind_rows(
    eliminadas_inf,
    data.frame(Variable = names(predictores_inf)[high_corr], VIF = NA)
  )
  predictores_inf <- predictores_inf[, -high_corr, drop = FALSE]
}

#  Filtrar variables por VIF 
vif_sel_inf <- vifstep(predictores_inf, th = 5)

vif_eliminadas_inf <- setdiff(names(predictores_inf), vif_sel_inf@results$Variables)
if (length(vif_eliminadas_inf) > 0) {
  eliminadas_inf <- bind_rows(
    eliminadas_inf,
    data.frame(Variable = vif_eliminadas_inf, VIF = ">5")
  )
}

# Crear dataset limpio para Lim_inf 
datos_limpios_inf <- datos_numericos_limpio_3[, c("Lim_inf", vif_sel_inf@results$Variables), drop = FALSE]

# Ajustar modelo final 
modelo_final_inf <- lm(Lim_inf ~ ., data = datos_limpios_inf)
